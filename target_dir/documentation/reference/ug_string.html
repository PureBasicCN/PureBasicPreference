用户指南 - 字符串操作

下面的示例将一步一步演示，字符串string库中的各种命令及他们的用途和正确的使用方法。 
  Define.s String1, String2, String3
  
  String1 = "The quick brown fox jumps over the lazy dog."
  
  ; ‘Left’从字符串的左边返回指定字长度的字符串.
  ; ‘Mid’返回字符串的中间起始位置开始，指定长度的字符串.
  ; ‘Right’从字符串的右端返回指定长度的字符串.
  ; ‘Space’返回指定数量的空格字符组成的字符串.
  ; 结果将输出 "The brown dog."
  Debug "* Left, Mid and Right"
  String2 = Left(String1, 3) + Space(1) + Mid(String1, 11, 5) + Space(1) + Right(String1, 4)
  Debug String2
  
  ;‘CountString’返回在第一个字符串中搜索到的第二个字符串的数量，区分大小写。
  ; 输出显示 1.
  Debug "* CountString"
  Debug CountString(String1, "the")
  
  ; 使用‘LCase’ (和‘UCase’)命令对指定字符串进行大（小）写转换。
  ; 输出显示 2
  Debug "* CountString and LCase"
  String2 = LCase(String1)
  Debug CountString(String2, "the")
  
  ; ‘FindString’命令用来在一个字符串中搜索另一个字符串的位置.
  ; 输出显示 17.
  Debug "* FindString"
  Debug FindString(String1, "fox")
  
  ; ‘RemoveString’用来删除字符串中指定的字串.
  ;显示：The quick fox jumps over the lazy dog.
  Debug "* RemoveString"
  String2 = RemoveString(String1, " brown")
  Debug String2
  
  ; ‘ReplaceString’用于替换字符串中内容。
  ; 显示 The quick brown fox jumps over the sleeping dog.
  Debug "* ReplaceString"
  String2 = ReplaceString(String1, "lazy", "sleeping")
  Debug String2
  
  ;‘StringByteLength’返回以字节为单位的字符串长度。
  ; 如果不指定编码(不包含：null).
  Debug "* StringByteLength"
  ; 显示 44.
  Debug StringByteLength(String1, #PB_Ascii)
  ; 显示 88.
  Debug StringByteLength(String1, #PB_Unicode)
  
  ; 'StringField'返回指定索引处的字符串字段。
  ; 此命令很有用，如将字符串转换为列表等等。
  ; 'StringField'可以将空格作为分隔符 
  ; 但本示例中使用更明显的'\'来作为分隔符。
  ; 显示 jumps.
  Debug "* StringField"
  String2 = ReplaceString(String1, " ", "\")
  Debug String2
  String3 = StringField(String2, 5, "\")
  Debug String3
  
  ;  'Trim'将指定字符串中不必要的空格删除掉。
  ; 同样，'LTrim'、'RTrim'分别删除指定字符串左（开头）侧或右侧（结尾）部分的多余的空格符。
  Debug "* Trim, LTrim and RTrim"
  String2 = Space(10) + String1 + Space(8)
  Debug #DQUOTE$ + String2 + #DQUOTE$
  String3 = Trim(String2)
  Debug #DQUOTE$ + String3 + #DQUOTE$
  String3 = LTrim(String2)
  Debug #DQUOTE$ + String3 + #DQUOTE$
  String3 = RTrim(String2)
  Debug #DQUOTE$ + String3 + #DQUOTE$
  
  ; 'LSet '将字符串设置成为从左端开始指定长度的字符串，不足时用空格填充
  ; 或用其他指定的字符填充.
  ;如果字符串已经超过指定长度，则将其截断。
  Debug "*LSet"
  Debug LSet("Abc", 10, "*")
  Debug LSet("Abcd", 10, "*")
  Debug LSet("Abcde", 10, "*")
  
  ; 同样，'RSet'是从字符串右侧开始向左指定长度。
  Debug "* RSet"
  Debug RSet("1.23", 10, "0")
  Debug RSet("10.23", 10, "0")
  Debug RSet("100.23", 10, "0")
  
  ; 'Str'将quad 型值转换成为字符型, 同样，'StrF'是转换浮点 floats型值,  
  ; 'StrD'是转换 doubles 型，'StrU'转换 unsigned 型, 这些函数都有可选参数 
  ; 用来指定要显示的十进制位数。
  Debug "* Str, StrF and StrD"
  Debug Str(100)
  Debug StrF(1.234, 3)
  Debug StrD(123456.789, 3)
  
  ;  'Val'将字符型值转换成数值型 (quad) .
   'ValD'和'ValF' 同样是将字符串转换成数值，但类型是浮点 floats型和 doubles型.
  Debug "* Val"
  Debug Val("123")
  
  ; 'Bin' 将数值转换成二进制型。
  ;  'Hex'将数值转换成十六进制型。
  Debug "* Bin and Hex"
  Debug Bin(19)
  Debug Hex(19)
